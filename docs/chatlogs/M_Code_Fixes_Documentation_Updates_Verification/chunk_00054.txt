"""
Verify December 2025 Overtime/TimeOff data:
1. Compare backfill months (12-24 through 11-25) between November and December exports
2. Extract and display 12-25 values for verification
"""

import csv
from pathlib import Path
from typing import Dict, Tuple

def load_export(path: Path) -> Dict[Tuple[str, str], float]:
    """Load LONG format export: (Time_Category, PeriodLabel) -> value"""
    data = {}
    with path.open("r", newline="", encoding="utf-8-sig") as f:
        reader = csv.DictReader(f)
        for row in reader:
            cat = (row.get("Time Category") or "").strip()
            period = (row.get("PeriodLabel") or "").strip()
            val_str = (row.get("Sum of Value") or "").strip()
            if cat and period and val_str:
                try:
                    data[(cat, period)] = float(val_str)
                except ValueError:
                    pass
    return data

def main():
    nov_path = Path("2025_11_Monthly Accrual and Usage Summary.csv")
    dec_path = Path("2026_01_12_18_48_07_Monthly Accrual and Usage Summary.csv")
    
    if not nov_path.exists():
        print(f"ERROR: {nov_path} not found")
        return 1
    if not dec_path.exists():
        print(f"ERROR: {dec_path} not found")
        return 1
    
    nov_data = load_export(nov_path)
    dec_data = load_export(dec_path)
    
    # Months to compare (backfill months)
    backfill_months = ["12-24", "01-25", "02-25", "03-25", "04-25", "05-25", 
                      "06-25", "07-25", "08-25", "09-25", "10-25", "11-25"]
    
    # Current month
    current_month = "12-25"
    
    print("=" * 80)
    print("BACKFILL DATA VERIFICATION (12-24 through 11-25)")
    print("=" * 80)
    print(f"November export: {nov_path}")
    print(f"December export: {dec_path}")
    print()
    
    # Get all categories
    all_cats = sorted(set(cat for cat, period in set(nov_data.keys()) | set(dec_data.keys()) 
                         if period in backfill_months + [current_month]))
    
    differences = []
    missing_in_nov = []
    missing_in_dec = []
    
    # Compare backfill months
    for cat in all_cats:
        for month in backfill_months:
            nov_val = nov_data.get((cat, month))
            dec_val = dec_data.get((cat, month))
            
            if nov_val is None and dec_val is not None:
                missing_in_nov.append((cat, month, dec_val))
            elif dec_val is None and nov_val is not None:
                missing_in_dec.append((cat, month, nov_val))
            elif nov_val is not None and dec_val is not None:
                if abs(nov_val - dec_val) > 1e-6:
