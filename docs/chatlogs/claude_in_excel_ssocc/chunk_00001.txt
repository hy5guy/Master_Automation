// 🕒 2026-02-14-12-00-00 (EST)
// SSOCC/SSOCC - Service Log.xlsx
// Author: R. A. Carucci
// Purpose: Create placeholder sheets and tables for future 2026 months

const monthMapping = [
  { sheetName: "MAR26", tableName: "_26_MAR" },
  { sheetName: "APR26", tableName: "_26_APR" },
  { sheetName: "MAY26", tableName: "_26_MAY" },
  { sheetName: "JUN26", tableName: "_26_JUN" },
  { sheetName: "JUL26", tableName: "_26_JUL" },
  { sheetName: "AUG26", tableName: "_26_AUG" },
  { sheetName: "SEP26", tableName: "_26_SEP" },
  { sheetName: "OCT26", tableName: "_26_OCT" },
  { sheetName: "NOV26", tableName: "_26_NOV" },
  { sheetName: "DEC26", tableName: "_26_DEC" }
];

const results = [];

for (const mapping of monthMapping) {
  try {
    // Create new sheet
    const sheet = context.workbook.worksheets.add(mapping.sheetName);
    
    // Set up header row matching the JAN26 structure
    const headerRange = sheet.getRange("A1:I1");
    headerRange.values = [["Control #", "Date", "Time", "CCBP", "Service/Product", "Description", "LIS", "LIC", "OCA/CAD#"]];
    
    // Format header
    headerRange.format.font.bold = true;
    headerRange.format.font.size = 14;
    headerRange.format.font.color = "#000000";
    headerRange.format.fill.color = "#FFFFFF";
    headerRange.format.horizontalAlignment = "Center";
    headerRange.format.borders.getItem("EdgeBottom").style = "Continuous";
    headerRange.format.borders.getItem("EdgeBottom").weight = "Medium";
    
    // Create table from A1:I1 (just headers for now)
    const table = sheet.tables.add("A1:I1", true);
    table.name = mapping.tableName;
    
    results.push({
      sheet: mapping.sheetName,
      table: mapping.tableName,
      status: "Created (empty)"
    });
    
  } catch (error) {
    results.push({
      sheet: mapping.sheetName,
      table: mapping.tableName,
      status: "Error",
      error: error.message
    });
  }
}

await context.sync();

return { tablesCreated: results };
